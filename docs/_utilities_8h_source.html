75  [[nodiscard]] [[deprecated("This implementation of PopCount is superseeded by std::popcount in C++20. This "
76  "will be removed when the language version is updated.")]] inline int32_t
77  PopCount(uint32_t value) noexcept
78  {
79  const uint32_t c1 = 0x55555555u;
80  const uint32_t c2 = 0x33333333u;
81  const uint32_t c3 = 0x0F0F0F0Fu;
82  const uint32_t c4 = 0x01010101u;
83 
84  value -= (value >> 1) & c1;
85  value = (value & c2) + ((value >> 2) & c2);
86  value = (((value + (value >> 4)) & c3) * c4) >> 24;
87 
88  return static_cast<int32_t>(value);
89  }
