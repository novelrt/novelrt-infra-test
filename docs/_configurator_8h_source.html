217  template<typename... TComponentTypes>
218  [[nodiscard]] SystemScheduler InitialiseAndRegisterComponents(
219  std::tuple<TComponentTypes, std::string>... deleteInstructionStatesAndSerialisedTypeNames)
220  {
221  SystemScheduler scheduler(_threadCount.value_or(0));
222 
223  if (_shouldAddDefaultSystems)
224  {
225  AddDefaultComponentsAndSystems(scheduler);
226  }
227 
228  for (auto&& system : _systems)
229  {
230  scheduler.RegisterSystem(system);
231  }
232 
233  for (auto&& iecsSystem : _iecsSystems)
234  {
235  scheduler.RegisterSystem(iecsSystem);
236  }
237 
238  scheduler.GetComponentCache().RegisterComponentType<TComponentTypes...>(
239  std::get<0>(deleteInstructionStatesAndSerialisedTypeNames)...,
240  std::get<1>(deleteInstructionStatesAndSerialisedTypeNames)...);
241  scheduler.SpinThreads();
242 
243  return scheduler;
244  }
